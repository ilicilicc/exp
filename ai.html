# Local LLM AI System - No API Keys Required ## Option 1: Using Ollama (Recommended for Beginners) ### Installation ```bash # Install Ollama curl -fsSL https://ollama.ai/install.sh | sh # Pull a model (examples) ollama pull llama2 ollama pull codellama ollama pull mistral ``` ### Python Interface ```python import requests import json class LocalAI: def __init__(self, model="llama2", base_url="http://localhost:11434"): self.model = model self.base_url = base_url def generate(self, prompt, max_tokens=150): url = f"{self.base_url}/api/generate" data = { "model": self.model, "prompt": prompt, "stream": False, "options": { "num_predict": max_tokens } } response = requests.post(url, json=data) if response.status_code == 200: return response.json()["response"] else: return f"Error: {response.status_code}" def chat(self, messages): url = f"{self.base_url}/api/chat" data = { "model": self.model, "messages": messages, "stream": False } response = requests.post(url, json=data) if response.status_code == 200: return response.json()["message"]["content"] else: return f"Error: {response.status_code}" # Usage example ai = LocalAI(model="llama2") response = ai.generate("Explain quantum computing in simple terms") print(response) ``` ## Option 2: Using Transformers Library (Hugging Face) ### Installation ```bash pip install transformers torch accelerate ``` ### Python Implementation ```python from transformers import AutoTokenizer, AutoModelForCausalLM import torch class HuggingFaceAI: def __init__(self, model_name="microsoft/DialoGPT-small"): self.tokenizer = AutoTokenizer.from_pretrained(model_name) self.model = AutoModelForCausalLM.from_pretrained(model_name) # Add padding token if it doesn't exist if self.tokenizer.pad_token is None: self.tokenizer.pad_token = self.tokenizer.eos_token def generate_response(self, prompt, max_length=100): inputs = self.tokenizer.encode(prompt, return_tensors='pt', padding=True) with torch.no_grad(): outputs = self.model.generate( inputs, max_length=max_length, num_return_sequences=1, temperature=0.7, do_sample=True, pad_token_id=self.tokenizer.pad_token_id ) response = self.tokenizer.decode(outputs[0], skip_special_tokens=True) return response[len(prompt):].strip() # Usage ai = HuggingFaceAI("microsoft/DialoGPT-medium") response = ai.generate_response("Hello, how are you?") print(response) ``` ## Option 3: Using GPT4All ### Installation ```bash pip install gpt4all ``` ### Python Implementation ```python from gpt4all import GPT4All class GPT4AllAI: def __init__(self, model_name="orca-mini-3b-gguf2-q4_0.gguf"): self.model = GPT4All(model_name) def generate(self, prompt, max_tokens=150): response = self.model.generate( prompt, max_tokens=max_tokens, temp=0.7, top_k=40, top_p=0.9 ) return response def chat_session(self): with self.model.chat_session(): while True: user_input = input("You: ") if user_input.lower() in ['quit', 'exit']: break response = self.model.generate(user_input, max_tokens=150) print(f"AI: {response}") # Usage ai = GPT4AllAI() response = ai.generate("What is artificial intelligence?") print(response) ``` ##
